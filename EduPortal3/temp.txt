function validateProfileForm() {
    const loginInput = document.querySelector('input[name="login"]');
    const emailInput = document.querySelector('input[name="email"]');
    const birthdateInput = document.querySelector('input[name="birth_date"]');
    let isValid = true;

    // –í–∞–ª–∏–¥–∞—Ü–∏—è –ª–æ–≥–∏–Ω–∞
    if (loginInput.value.length < 3 || loginInput.value.length > 20) {
        loginInput.classList.add('is-invalid');
        isValid = false;
    } else {
        loginInput.classList.remove('is-invalid');
    }

    // –í–∞–ª–∏–¥–∞—Ü–∏—è email
    const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailPattern.test(emailInput.value)) {
        emailInput.classList.add('is-invalid');
        isValid = false;
    } else {
        emailInput.classList.remove('is-invalid');
    }

    // –í–∞–ª–∏–¥–∞—Ü–∏—è –¥–∞—Ç—ã —Ä–æ–∂–¥–µ–Ω–∏—è
    const datePattern = /^\d{4}-\d{2}-\d{2}$/;
    if (!datePattern.test(birthdateInput.value)) {
        birthdateInput.classList.add('is-invalid');
        isValid = false;
    } else {
        birthdateInput.classList.remove('is-invalid');
    }

    return isValid;
}

    def wrapper(fn):

        def decorated_view(*args, **kwargs):

@app.route('/upload-avatar', methods=['POST'])
@login_required
def upload_avatar():
    if 'avatar' not in request.files:
        flash('–§–∞–π–ª –Ω–µ –≤—ã–±—Ä–∞–Ω', 'error')
        return redirect(url_for('profile'))

    file = request.files['avatar']
    if file.filename == '':
        flash('–§–∞–π–ª –Ω–µ –≤—ã–±—Ä–∞–Ω', 'error')
        return redirect(url_for('profile'))

    if file and allowed_file(file.filename):
        try:
            # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —É–Ω–∏–∫–∞–ª—å–Ω–æ–µ –∏–º—è —Ñ–∞–π–ª–∞
            filename = f"{current_user.id}_{int(time.time())}.{file.filename.rsplit('.', 1)[1].lower()}"
            filepath = os.path.join(current_app.config['UPLOAD_FOLDER'], filename)

            # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Ñ–∞–π–ª
            file.save(filepath)

            # –û–±–Ω–æ–≤–ª—è–µ–º –∑–∞–ø–∏—Å—å –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
            users = get_users()
            for user in users:
                if user['id'] == current_user.id:
                    # –£–¥–∞–ª—è–µ–º —Å—Ç–∞—Ä—ã–π –∞–≤–∞—Ç–∞—Ä
                    if user.get('profile_image'):
                        old_path = os.path.join(current_app.config['UPLOAD_FOLDER'], user['profile_image'])
                        if os.path.exists(old_path):
                            try:
                                os.remove(old_path)
                            except Exception as e:
                                app.logger.error(f"Error deleting old avatar: {str(e)}")

                    user['profile_image'] = filename
                    break

            save_users(users)
            flash('–ê–≤–∞—Ç–∞—Ä —É—Å–ø–µ—à–Ω–æ –æ–±–Ω–æ–≤–ª–µ–Ω!', 'success')
        except Exception as e:
            app.logger.error(f"Avatar upload error: {str(e)}")
            flash('–û—à–∏–±–∫–∞ –ø—Ä–∏ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏–∏ —Ñ–∞–π–ª–∞', 'error')
    else:
        flash('–†–∞–∑—Ä–µ—à–µ–Ω—ã —Ç–æ–ª—å–∫–æ —Ñ–∞–π–ª—ã: PNG, JPG, JPEG, GIF', 'error')

    return redirect(url_for('profile'))


@app.route('/profile', methods=['GET', 'POST'])
@login_required

        try:

                datetime.strptime(new_data['birth_date'], '%Y-%m-%d')

            users = get_users()
            for user in users:
                if user['id'] != current_user.id:
                    if user['login'] == new_data['login']:
                        raise ValueError('–≠—Ç–æ—Ç –ª–æ–≥–∏–Ω —É–∂–µ –∑–∞–Ω—è—Ç')

            for user in users:
                if user['id'] == current_user.id:
                    user.update(new_data)

def create_user(login, email, password, full_name, birth_date, account_type, school=None):
    users = get_users()

    # –ü—Ä–æ–≤–µ—Ä–∫–∞ —É–Ω–∏–∫–∞–ª—å–Ω–æ—Å—Ç–∏ –ª–æ–≥–∏–Ω–∞ –∏ email
    if any(u['login'] == login for u in users):
        return None, '–õ–æ–≥–∏–Ω —É–∂–µ –∑–∞–Ω—è—Ç'
    if any(u['email'] == email for u in users):
        return None, 'Email —É–∂–µ –∑–∞—Ä–µ–≥–∏—Å—Ç—Ä–∏—Ä–æ–≤–∞–Ω'

    new_user = {
        'id': max([u['id'] for u in users], default=0) + 1,
        'login': login,
        'email': email,
        'password_hash': generate_password_hash(password),
        'full_name': full_name,
        'school': school,
        'birth_date': birth_date,
        'account_type': account_type,
        'created_at': datetime.now().isoformat()
    }

    users.append(new_user)
    save_users(users)
    return new_user, None

def delete_message_from_db(message_id):
    try:
        print(f"\nAttempting to delete message {message_id}")
        data = load_forum_data()
        initial_count = len(data['messages'])
        print(f"Initial messages count: {initial_count}")

        # –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è —Å–æ–æ–±—â–µ–Ω–∏–π
        data['messages'] = [msg for msg in data['messages'] if str(msg['id']) != str(message_id)]
        new_count = len(data['messages'])
        print(f"New messages count: {new_count}")

        if new_count == initial_count:
            print("Message not found in database")
            return False

        # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ
        save_forum_data(data)
        print("Message successfully deleted")
        return True

    except Exception as e:
        print(f"Deletion error: {str(e)}")
        return False

def get_solved_tasks(user_id):
    """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ª–æ–≤–∞—Ä—å {task_id: is_correct} –¥–ª—è –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–≥–æ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è."""
    tasks = load_tasks()
    solved_tasks = {}
    for task in tasks:
        if 'solved_by' in task:
            for solver in task['solved_by']:
                if solver['user_id'] == user_id:
                    solved_tasks[task['id']] = solver['is_correct']
    return solved_tasks


def save_solved_task(user_id, task_id, is_correct):
    """–°–æ—Ö—Ä–∞–Ω—è–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç —Ä–µ—à–µ–Ω–∏—è –∑–∞–¥–∞—á–∏."""
    tasks = load_tasks()
    for task in tasks:
        if task['id'] == task_id:
            if 'solved_by' not in task:
                task['solved_by'] = []

            # –£–¥–∞–ª—è–µ–º —Å—Ç–∞—Ä—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç, –µ—Å–ª–∏ –æ–Ω –µ—Å—Ç—å
            task['solved_by'] = [s for s in task['solved_by'] if s['user_id'] != user_id]

            # –î–æ–±–∞–≤–ª—è–µ–º –Ω–æ–≤—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç
            task['solved_by'].append({
                'user_id': user_id,
                'is_correct': is_correct,
                'timestamp': datetime.now().isoformat()
            })
            break

    save_tasks(tasks)
